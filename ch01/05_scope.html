① グローバル変数とローカル変数

重要ポイント：スコープとは「変数が参照できる範囲」のことです。
関数の外で宣言した変数はグローバル変数（どこからでも参照可能）、 関数の中で宣言した変数はローカル変数（その関数の中でのみ参照可能）になります。

🗺️ スコープ可視化マップ（変数をクリックしてアクセス可否を確認）

🌐 グローバルスコープ どこからでもアクセス可
globalX = 100
globalMsg = "Hello"
⚙️ 関数スコープ（outer） outer内のみ
outerVar = 200
⚙️ 関数スコープ（inner） inner内のみ
innerVar = 300
inner からアクセスできる変数： innerVar、 outerVar、 globalX、 globalMsg
outer からアクセスできる変数： outerVar、 globalX、 globalMsg （innerVar にはアクセス不可） 
// グローバル変数（関数の外で宣言）
let globalX   = 100;
let globalMsg  = "Hello";

function outer() {
  // ローカル変数（outer 関数内で宣言）
  let outerVar = 200;

  // グローバル変数にアクセスできる
  console.log(globalX);   // → 100 ✅
  console.log(outerVar);  // → 200 ✅

  function inner() {
    // ローカル変数（inner 関数内で宣言）
    let innerVar = 300;

    // 外側のすべての変数にアクセスできる
    console.log(globalX);  // → 100 ✅
    console.log(outerVar); // → 200 ✅
    console.log(innerVar); // → 300 ✅
  }
  inner();

  // inner のローカル変数にはアクセスできない
  console.log(innerVar); // → ReferenceError ❌
}

outer();
// グローバルからローカル変数にはアクセスできない
console.log(outerVar); // → ReferenceError ❌

② スコープチェーン

重要ポイント：変数を参照するとき、JavaScriptは現在のスコープから外側へ順番に変数を探します。これをスコープチェーンといいます。
見つかった時点で探索を止め、最終的にグローバルスコープまで見つからなければ ReferenceError になります。

スコープチェーンの探索順序

🔍 現在のスコープ（inner 関数）
innerVar = 300
x = 30
↑ 見つからなければ外側へ
🔍 外側のスコープ（outer 関数）
outerVar = 200
x = 20
↑ 見つからなければ外側へ
🔍 グローバルスコープ
globalX = 100
x = 10
↑ ここまで見つからなければ ReferenceError
← 下のシミュレーターで変数を検索してみましょう
スコープチェーンの探索をシミュレートしよう
を inner から参照
← 変数を選んで探索してみましょう
// スコープチェーンの例
let x = 10; // グローバル

function outer() {
  let x = 20; // outer のローカル（グローバルを隠す）

  function inner() {
    let x = 30; // inner のローカル（outer を隠す）
    console.log(x); // → 30（最も近いスコープの x）
  }

  inner();
  console.log(x); // → 20（outer のローカル x）
}

outer();
console.log(x); // → 10（グローバル x）

// 変数シャドーイング：内側の変数が外側の同名変数を「隠す」
let name = "グローバル";
function test() {
  let name = "ローカル"; // グローバルの name を隠す（シャドーイング）
  console.log(name);    // → "ローカル"
}
test();
console.log(name);      // → "グローバル"（影響なし）

③ let と var の違い

重要ポイント：let はブロックスコープ、var は関数スコープです。この違いがバグの原因になりやすいため、現代のJavaScriptでは let / const を使います。

スコープの違いを比較

✅ let（ブロックスコープ）

📦 { } ブロック内でのみ有効
🔒 同じスコープで再宣言不可
⏰ TDZ（宣言前アクセス不可）
🌍 グローバル汚染なし

⚠️ var（関数スコープ）

📦 関数全体で有効（ブロック無視）
🔓 同じスコープで再宣言可能
⬆️ ホイスティング（undefined）
🌍 グローバル宣言で window に追加

ブロックスコープの違い

// ===== let：ブロックスコープ =====
{
  let blockLet = "let の値";
  console.log(blockLet); // → "let の値" ✅
}
console.log(blockLet); // → ReferenceError ❌（ブロック外は参照不可）

// ===== var：ブロックスコープを無視 =====
{
  var blockVar = "var の値";
  console.log(blockVar); // → "var の値" ✅
}
console.log(blockVar); // → "var の値" ✅（ブロック外でも参照できる！）

// if 文のブロックでも同様
if (true) {
  let letInIf = "let";
  var varInIf = "var";
}
// console.log(letInIf); // → ReferenceError
console.log(varInIf);   // → "var"（if の外でも使える）
⚠️ var はブロックスコープを無視するため、意図しない場所で変数が参照できてしまいます。これがバグの原因になりやすい理由です。

ループでの let vs var（試験頻出！）

// ===== var を使ったループ（よくあるバグ）=====
for (var i = 0; i < 3; i++) {
  setTimeout(() => {
    console.log(i); // var は関数スコープなので i が共有される
  }, 100);
}
// → 3, 3, 3（すべて 3 になる！）
// ループ終了後に i = 3 になっているため

// ===== let を使ったループ（正しい動作）=====
for (let j = 0; j < 3; j++) {
  setTimeout(() => {
    console.log(j); // let はブロックスコープなので各ループで独立
  }, 100);
}
// → 0, 1, 2（期待通りの動作）
// 各ループで j の新しいバインディングが作られるため

ホイスティングの違い

// ===== var のホイスティング =====
console.log(varX); // → undefined（エラーにならない！）
var varX = 10;
console.log(varX); // → 10

// 上のコードは内部的にこう解釈される
var varX;           // 宣言だけが先頭に「巻き上げ」られる
console.log(varX); // → undefined
varX = 10;          // 代入はその場で実行
console.log(varX); // → 10

// ===== let のホイスティング（TDZ）=====
console.log(letX); // → ReferenceError（TDZ: 一時的なデッドゾーン）
let letX = 10;
console.log(letX); // → 10

// let も宣言は巻き上げられるが、初期化前はアクセス不可（TDZ）
// TDZ = Temporal Dead Zone（一時的なデッドゾーン）
✅ TDZ（Temporal Dead Zone）のメリット
let / const の TDZ は、宣言前のアクセスを明確なエラーにします。 var のように undefined が返るよりも、バグを早期に発見できます。

クロージャ入門

クロージャとは：関数が定義されたスコープの変数を、そのスコープが終わった後も参照し続ける仕組みです。 
// クロージャの基本例
function makeCounter() {
  let count = 0; // この変数がクロージャで保持される

  return function() {
    count++;
    return count;
  };
}

const counter1 = makeCounter();
const counter2 = makeCounter(); // 独立したカウンター

console.log(counter1()); // → 1
console.log(counter1()); // → 2
console.log(counter1()); // → 3
console.log(counter2()); // → 1（counter2 は独立）

// クロージャを使ったプライベート変数
function createBankAccount(initialBalance) {
  let balance = initialBalance; // 外からは直接アクセスできない

  return {
    deposit: (amount) => { balance += amount; return balance; },
    withdraw: (amount) => { balance -= amount; return balance; },
    getBalance: () => balance
  };
}

const account = createBankAccount(1000);
console.log(account.deposit(500));    // → 1500
console.log(account.withdraw(200));   // → 1300
console.log(account.getBalance());   // → 1300

📌 まとめ

概念説明ポイント
グローバルスコープ 関数・ブロックの外で宣言した変数 どこからでもアクセス可能。汚染に注意
ローカルスコープ 関数内で宣言した変数 その関数内のみアクセス可能
ブロックスコープ { } 内で let/const で宣言した変数 そのブロック内のみアクセス可能
スコープチェーン 変数を内側から外側へ順番に探す仕組み 見つかった時点で探索終了
変数シャドーイング 内側の同名変数が外側の変数を隠す 内側の変数が優先される
TDZ let/const の宣言前アクセス禁止ゾーン ReferenceError が発生する
クロージャ 外側スコープの変数を保持し続ける関数 プライベート変数・カウンターなどに活用
