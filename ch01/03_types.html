🔍 typeof チェッカー（値を入力して型を確認）

← 値を入力して型を確認してみましょう 
// typeof 演算子で型を確認する
console.log(typeof "Hello");     // → "string"
console.log(typeof 42);          // → "number"
console.log(typeof 3.14);        // → "number"（整数も小数も同じ型）
console.log(typeof true);        // → "boolean"
console.log(typeof undefined);   // → "undefined"
console.log(typeof null);        // → "object" ← 歴史的バグ！
console.log(typeof {});          // → "object"
console.log(typeof []);          // → "object"（配列も object）
console.log(typeof function(){}); // → "function"
console.log(typeof Symbol());     // → "symbol"
console.log(typeof 9007199254740993n); // → "bigint"
⚠️ typeof null === "object" は歴史的バグです
null の typeof は "object" を返しますが、null はオブジェクトではありません。 これはJavaScriptの最初の実装から残るバグで、後方互換性のために修正されていません。 null チェックは value === null で行いましょう。

② プリミティブ型

重要ポイント：string・number・boolean・undefined・null・symbol・bigint の7種類がプリミティブ型です。
プリミティブ型は値そのものがコピーされます（値渡し）。オブジェクト型は参照がコピーされます（参照渡し）。

値渡し vs 参照渡し（重要！）

プリミティブ型は「値」がコピーされる

// プリミティブ型：値がコピーされる
let a = 10;
let b = a;   // a の「値 10」がコピーされる
b = 20;      // b を変えても a は変わらない

console.log(a); // → 10（変わっていない）
console.log(b); // → 20

// 文字列も同様
let s1 = "Hello";
let s2 = s1;
s2 = "World";
console.log(s1); // → "Hello"（変わっていない）
console.log(s2); // → "World"
✅ プリミティブ型はコピーすると完全に独立した値になります。

オブジェクト型は「参照」がコピーされる

// オブジェクト型：参照（アドレス）がコピーされる
const obj1 = { name: "Alice", age: 25 };
const obj2 = obj1;  // 同じオブジェクトを指す参照がコピーされる

obj2.name = "Bob";  // obj2 を変更すると…
console.log(obj1.name); // → "Bob"（obj1 も変わる！）
console.log(obj2.name); // → "Bob"

// 配列も同様（配列はオブジェクトの一種）
const arr1 = [1, 2, 3];
const arr2 = arr1;
arr2.push(4);
console.log(arr1); // → [1, 2, 3, 4]（arr1 も変わる！）

// コピーしたい場合はスプレッド構文を使う
const obj3 = { ...obj1 };  // シャローコピー
const arr3 = [...arr1];    // シャローコピー
obj3.name = "Carol";
console.log(obj1.name); // → "Bob"（変わらない）
⚠️ オブジェクトを別の変数に代入しても、同じオブジェクトを参照しています。 独立したコピーを作るにはスプレッド構文 { ...obj } や [...arr] を使いましょう。

型変換（明示的・暗黙的）

// ===== 明示的な型変換 =====

// 数値 → 文字列
console.log(String(42));        // → "42"
console.log((42).toString());   // → "42"
console.log((255).toString(16)); // → "ff"（16進数）

// 文字列 → 数値
console.log(Number("42"));      // → 42
console.log(Number("3.14"));    // → 3.14
console.log(Number(""));        // → 0
console.log(Number("abc"));     // → NaN
console.log(parseInt("42px"));  // → 42（先頭の数値部分のみ）
console.log(parseFloat("3.14rem")); // → 3.14

// 真偽値への変換
console.log(Boolean(0));        // → false（falsy）
console.log(Boolean(""));       // → false（falsy）
console.log(Boolean(null));     // → false（falsy）
console.log(Boolean(1));        // → true（truthy）
console.log(Boolean("hello"));  // → true（truthy）

// ===== 暗黙的な型変換（注意！）=====
console.log("5" - 2);   // → 3  （- は数値変換）
console.log("5" * 2);   // → 10 （* は数値変換）
console.log("5" + 2);   // → "52"（+ は文字列連結！）

③ 配列

重要ポイント：配列は複数の値を順番に格納するデータ構造です。インデックスは 0 から始まります。

配列ビジュアライザ

配列を操作してビジュアルで確認しよう
← 操作を選んで実行してみましょう
// 配列の作成
const fruits = ["apple", "banana", "cherry"];

// インデックスでアクセス（0始まり）
console.log(fruits[0]); // → "apple"
console.log(fruits[2]); // → "cherry"
console.log(fruits.length); // → 3

// 末尾への追加・削除
fruits.push("date");    // → ["apple","banana","cherry","date"]
fruits.pop();           // → ["apple","banana","cherry"]（"date"を削除）

// 先頭への追加・削除
fruits.unshift("avocado"); // → ["avocado","apple","banana","cherry"]
fruits.shift();              // → ["apple","banana","cherry"]

// 部分取得（元の配列は変更しない）
console.log(fruits.slice(0, 2)); // → ["apple","banana"]

// 検索
console.log(fruits.indexOf("banana"));  // → 1
console.log(fruits.includes("cherry")); // → true

// splice：要素の追加・削除・置換
fruits.splice(1, 1, "blueberry"); // インデックス1から1個削除して"blueberry"を挿入
console.log(fruits); // → ["apple","blueberry","cherry"]

④ オブジェクト

重要ポイント：オブジェクトはキーと値のペアでデータを管理します。ドット記法またはブラケット記法でアクセスします。

オブジェクトビジュアライザ

{
name : "Alice" ,
age : 25 ,
city : "Tokyo" ,
active : true
}

インタラクティブ：オブジェクトを操作しよう

プロパティを操作してビジュアルで確認しよう
← 操作を選んで実行してみましょう
// オブジェクトの作成
const user = {
  name:   "Alice",
  age:    25,
  city:   "Tokyo",
  active: true
};

// プロパティへのアクセス（ドット記法）
console.log(user.name);   // → "Alice"
console.log(user.age);    // → 25

// ブラケット記法（変数でキーを指定できる）
const key = "name";
console.log(user[key]);    // → "Alice"

// プロパティの変更・追加・削除
user.age = 26;             // 変更
user.email = "a@example.com"; // 追加
delete user.active;       // 削除

// キー・値・エントリの取得
console.log(Object.keys(user));    // → ["name","age","city","email"]
console.log(Object.values(user));  // → ["Alice",26,"Tokyo","a@example.com"]
console.log(Object.entries(user)); // → [["name","Alice"],["age",26],...]

// プロパティの存在確認
console.log("name" in user);       // → true
console.log(user.hasOwnProperty("age")); // → true

// スプレッド構文でコピー・マージ
const copy   = { ...user };
const merged = { ...user, role: "admin" };

分割代入（Destructuring）

// オブジェクトの分割代入
const user = { name: "Alice", age: 25, city: "Tokyo" };
const { name, age } = user;
console.log(name); // → "Alice"
console.log(age);  // → 25

// 別名をつける
const { name: userName, age: userAge } = user;
console.log(userName); // → "Alice"

// デフォルト値を設定
const { name: n, score = 0 } = user;
console.log(score); // → 0（user に score がないのでデフォルト値）

// 配列の分割代入
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first);  // → 1
console.log(second); // → 2
console.log(rest);   // → [3, 4, 5]
