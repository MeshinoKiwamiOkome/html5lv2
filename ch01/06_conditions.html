① if 文

重要ポイント：if 文は条件が true のときだけ処理を実行します。 else if・else を組み合わせて複数の条件を処理できます。

🎮 if 文シミュレーター

数値を入力して if / else if / else の分岐を確認しよう
← 点数を入力して判定してみましょう
// if / else if / else の基本
let score = 75;

if (score >= 90) {
  console.log("優秀");
} else if (score >= 70) {
  console.log("良好");   // → "良好"（75 >= 70 が true）
} else if (score >= 50) {
  console.log("合格");
} else {
  console.log("不合格");
}

// 早期リターンパターン（ネストを減らす）
function getGrade(score) {
  if (score < 0 || score > 100) return "無効";
  if (score >= 90) return "A";
  if (score >= 70) return "B";
  if (score >= 50) return "C";
  return "D";
}
console.log(getGrade(85)); // → "B"

③ 論理演算子

重要ポイント：&&（AND）・||（OR）・!（NOT）の3種類があります。 短絡評価（ショートサーキット）の動作を理解することが重要です。

&&（AND）：両方 true のとき true

// && ：左辺と右辺が両方 true のとき true
console.log(true  && true);  // → true
console.log(true  && false); // → false
console.log(false && true);  // → false
console.log(false && false); // → false

// 実用例：複数条件のチェック
const age     = 20;
const hasCard = true;

if (age >= 18 && hasCard) {
  console.log("入場できます"); // → "入場できます"
}

// && の短絡評価：左辺が false なら右辺を評価しない
const user = null;
const name = user && user.name; // user が null なので user.name は評価されない
console.log(name); // → null（エラーにならない）

||（OR）：どちらか一方が true のとき true

// || ：どちらか一方が true のとき true
console.log(true  || true);  // → true
console.log(true  || false); // → true
console.log(false || true);  // → true
console.log(false || false); // → false

// 実用例：デフォルト値の設定
const input = "";
const name  = input || "ゲスト"; // input が falsy なら "ゲスト"
console.log(name); // → "ゲスト"

// || の短絡評価：左辺が true なら右辺を評価しない
const val = 5 || 10;
console.log(val); // → 5（左辺が truthy なので右辺は評価されない）

const val2 = 0 || 10;
console.log(val2); // → 10（左辺が falsy なので右辺を返す）

!（NOT）：真偽値を反転する

// ! ：true → false、false → true
console.log(!true);   // → false
console.log(!false);  // → true
console.log(!0);      // → true（0 は falsy）
console.log(!"");     // → true（空文字は falsy）
console.log(!"hello");// → false（文字列は truthy）
console.log(!null);   // → true（null は falsy）

// !! ：値を boolean に変換するテクニック
console.log(!!0);      // → false
console.log(!!"hello");// → true
console.log(!!null);   // → false
console.log(!![]);     // → true（空配列は truthy！）

// 実用例
const isLoggedIn = !(user === null);
const isEmpty    = !arr.length;

短絡評価（ショートサーキット）

短絡評価とは：論理演算子が結果を確定できた時点で、残りの評価を行わない仕組みです。 パフォーマンス向上だけでなく、エラー防止にも活用できます。 
// && の短絡評価：左辺が falsy なら右辺を評価しない
const user = null;
const name = user && user.name; // user が null → user.name は評価されない
console.log(name); // → null（TypeError にならない）

// オプショナルチェーン（?.）でも同様のことができる（ES2020〜）
const name2 = user?.name;
console.log(name2); // → undefined

// || の短絡評価：左辺が truthy なら右辺を評価しない
const config = null;
const timeout = config || 3000; // config が null → 3000 を使う
console.log(timeout); // → 3000

// 条件付きの関数呼び出し
const callback = null;
callback && callback(); // callback が null なので呼び出されない（エラーなし）

// JSX風の条件付きレンダリング（Reactなどで頻出）
const isAdmin = true;
const element = isAdmin && "管理者メニュー";
console.log(element); // → "管理者メニュー"

??（Nullish 合体演算子）ES2020〜

|| との違い：|| はすべての falsy 値（0・""・false など）に反応しますが、 ?? は null と undefined のみに反応します。 
// ?? ：null または undefined のときだけ右辺を返す
console.log(null      ?? "デフォルト"); // → "デフォルト"
console.log(undefined ?? "デフォルト"); // → "デフォルト"
console.log(0         ?? "デフォルト"); // → 0（0 は null/undefined ではない）
console.log(""        ?? "デフォルト"); // → ""（空文字も null/undefined ではない）
console.log(false     ?? "デフォルト"); // → false

// || との比較
const count = 0;
console.log(count || 10);  // → 10（0 は falsy なので）← 意図しない場合がある
console.log(count ?? 10);  // → 0（0 は null/undefined ではない）← 意図通り

// ?.（オプショナルチェーン）との組み合わせ
const user = null;
const city = user?.address?.city ?? "不明";
console.log(city); // → "不明"
|| と ?? の違いを確認しよう
← 値を入力して || と ?? の違いを確認しましょう

④ 条件演算子（三項演算子）

重要ポイント：条件 ? 真の値 : 偽の値 という構文で、if 文を1行で書けます。 シンプルな条件分岐に適していますが、複雑な条件には if 文を使いましょう。 
// 三項演算子の基本
// 条件 ? 真のとき : 偽のとき
const age   = 20;
const label = age >= 18 ? "成人" : "未成年";
console.log(label); // → "成人"

// if 文と同等
let result;
if (age >= 18) {
  result = "成人";
} else {
  result = "未成年";
}

// 変数への代入・テンプレートリテラルとの組み合わせ
const score = 85;
console.log(`結果：${score >= 60 ? "合格" : "不合格"}`);
// → "結果：合格"

// ネストした三項演算子（可読性が下がるので注意）
const grade = score >= 90 ? "A"
            : score >= 70 ? "B"
            : score >= 50 ? "C"
            :               "D";
console.log(grade); // → "B"

⑤ switch 文

重要ポイント：switch 文は1つの値を複数の候補と比較するときに使います。 各 case の末尾に break を忘れると次の case に「落ちて」しまいます（フォールスルー）。

🎮 switch 文シミュレーター

case "monday": console.log("月曜日")
case "tuesday": console.log("火曜日")
case "wednesday": console.log("水曜日")
case "saturday": console.log("土曜日（休日）")
case "sunday": console.log("日曜日（休日）")
default: console.log("平日") 
// switch 文の基本
const day = "monday";

switch (day) {
  case "monday":
    console.log("月曜日");
    break; // ← break を忘れると次の case に「落ちる」
  case "tuesday":
    console.log("火曜日");
    break;
  case "saturday":
  case "sunday":
    console.log("休日"); // 複数の case をまとめる（フォールスルーの活用）
    break;
  default:
    console.log("平日");
}

// break を忘れた場合（フォールスルー）
switch ("a") {
  case "a":
    console.log("A"); // ← break なし
  case "b":
    console.log("B"); // ← ここも実行される！
  case "c":
    console.log("C"); // ← ここも実行される！
    break;
}
// → "A", "B", "C" がすべて出力される

⑥ JavaScriptにおけるクセのある真偽値判定

重要ポイント：JavaScriptでは boolean 以外の値も条件式で使えます。 false と同等に扱われる値を「falsy」、true と同等に扱われる値を「truthy」といいます。
特に 0・""・[]・{} の扱いは試験頻出です。

falsy / truthy 一覧

❌ falsy な値（6種類のみ）

false false
0 false
-0 false
""（空文字） false
null false
undefined false
NaN false

✅ truthy な値（上記以外すべて）

true true
1（0以外の数値） true
"hello"（空でない文字列） true
"0"（文字列の "0"）⚠️ true
[]（空配列）⚠️ true
{}（空オブジェクト）⚠️ true
function(){} true
⚠️ 試験頻出の落とし穴
・"0"（文字列の "0"）は truthy です（数値の 0 は falsy）
・[]（空配列）は truthy です
・{}（空オブジェクト）は truthy です
・-0 は falsy です

🎮 truthy / falsy チェッカー

値を入力して truthy / falsy を確認しよう
← 値を入力して判定してみましょう
// falsy な値の確認
if (0)         console.log("truthy"); else console.log("falsy"); // → falsy
if ("")        console.log("truthy"); else console.log("falsy"); // → falsy
if (null)      console.log("truthy"); else console.log("falsy"); // → falsy
if (undefined) console.log("truthy"); else console.log("falsy"); // → falsy
if (NaN)       console.log("truthy"); else console.log("falsy"); // → falsy

// 注意！これらは truthy
if ("0")  console.log("truthy"); // → truthy（文字列の "0"）
if ([])   console.log("truthy"); // → truthy（空配列）
if ({})   console.log("truthy"); // → truthy（空オブジェクト）

// == を使うと予期しない比較になる
console.log(0  == false);  // → true（型変換が起きる）
console.log("" == false);  // → true
console.log("" == 0);      // → true
console.log(null == undefined); // → true
console.log(null === undefined);// → false（=== は型も比較）

📌 まとめ

構文 / 演算子説明ポイント
if / else if / else 条件による分岐 早期リターンでネストを減らす
=== / !== 厳密等価・不等価（型も比較） 現代のJSでは必ず === を使う
== / != 緩い等価・不等価（型変換あり） 予期しない結果になりやすい
&& AND：両方 true のとき true 左辺 falsy で短絡評価
|| OR：どちらか true のとき true 左辺 truthy で短絡評価・デフォルト値に活用
! NOT：真偽値を反転 !! で boolean 変換
?? Nullish 合体：null/undefined のみ反応 0 や "" を有効な値として扱える
条件 ? 真 : 偽 三項演算子 シンプルな条件分岐を1行で
switch / case / break 多岐分岐 break を忘れるとフォールスルー
