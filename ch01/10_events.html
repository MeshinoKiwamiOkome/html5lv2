① addEventListener

重要ポイント：addEventListener(イベント名, コールバック) で 要素にイベントリスナーを登録します。 同じ要素に複数のリスナーを登録でき、removeEventListener で解除できます。

🎮 イベントリスナー ライブデモ

↓ 各ボタンを操作してイベントを発火させよう

📋 イベントログ（最新10件）
← ボタンを操作するとここにイベントが表示されます
// addEventListener の基本
const btn = document.getElementById("myBtn");

// クリックイベント
btn.addEventListener("click", (event) => {
  console.log("クリックされました！");
  console.log(event.type);    // → "click"
  console.log(event.target);  // → クリックされた要素
  console.log(event.clientX, event.clientY); // → マウス座標
});

// 複数のイベントを登録できる
btn.addEventListener("mouseenter", () => console.log("ホバー開始"));
btn.addEventListener("mouseleave", () => console.log("ホバー終了"));

// イベントリスナーの解除
const handler = () => console.log("1回だけ");
btn.addEventListener("click", handler);
btn.removeEventListener("click", handler); // 解除

// once オプション：1回だけ実行
btn.addEventListener("click", () => console.log("1回だけ"), { once: true });

主要なイベント一覧

カテゴリイベント名発火タイミング

② イベントオブジェクトとイベント伝播

重要ポイント：イベントハンドラの引数として渡される event オブジェクトには イベントの詳細情報が含まれます。 イベントはデフォルトで内側から外側へ「バブリング」します。

🎮 イベント伝播（バブリング）デモ

🔵 outer（外側）
🟣 middle（中間）
🟢 inner（内側）
← ここをクリック
← 要素をクリックして伝播を確認しましょう
// イベントオブジェクトの主要プロパティ
document.addEventListener("click", (e) => {
  console.log(e.type);          // → "click"（イベント種類）
  console.log(e.target);        // → クリックされた要素
  console.log(e.currentTarget); // → リスナーが登録された要素
  console.log(e.clientX, e.clientY); // → マウス座標
  console.log(e.key);           // → キーボードのキー名
});

// バブリング：内側 → 外側へイベントが伝播
// inner クリック → inner → middle → outer → document の順に発火

// stopPropagation：伝播を止める
inner.addEventListener("click", (e) => {
  e.stopPropagation(); // 外側への伝播をここで止める
  console.log("inner のみ");
});

// preventDefault：デフォルト動作をキャンセル
link.addEventListener("click", (e) => {
  e.preventDefault(); // リンクのページ遷移をキャンセル
  console.log("遷移しない");
});

// イベントデリゲーション（親要素でまとめて処理）
ul.addEventListener("click", (e) => {
  if (e.target.tagName === "LI") {
    console.log(e.target.textContent);
  }
});

③ setTimeout / setInterval

重要ポイント：setTimeout は指定時間後に1回だけ実行し、 setInterval は指定間隔で繰り返し実行します。 どちらも非同期で実行されます。

🎮 タイマーデモ

setTimeout：指定時間後に1回だけ実行
ms 後に実行
← 時間とメッセージを設定して実行してみましょう
setInterval：指定間隔で繰り返し実行
ms 間隔
← 間隔を設定してカウンターを開始してみましょう
// setTimeout：指定時間後に1回だけ実行
const timerId = setTimeout(() => {
  console.log("2秒後に実行！");
}, 2000); // ミリ秒で指定

// キャンセル
clearTimeout(timerId);

// setInterval：指定間隔で繰り返し実行
let count = 0;
const intervalId = setInterval(() => {
  count++;
  console.log(`カウント: ${count}`);
  if (count >= 5) clearInterval(intervalId); // 5回で停止
}, 1000);

// 停止
clearInterval(intervalId);

// setTimeout(fn, 0) は「現在の処理が終わってから実行」
console.log("1番目");
setTimeout(() => console.log("3番目（非同期）"), 0);
console.log("2番目");
// → "1番目", "2番目", "3番目（非同期）"

④ Promise と async/await

重要ポイント：非同期処理の結果を扱うために Promise を使います。 async/await を使うと Promise を同期的なコードのように書けます。

Promise の状態遷移

⏳ pending
（待機中）
→
✅ fulfilled
（成功）
❌ rejected
（失敗）
→
.then(fn)
成功時の処理
.catch(fn)
失敗時の処理
→
.finally(fn)
常に実行

Promise の基本

// Promise の作成
const promise = new Promise((resolve, reject) => {
  // 非同期処理
  setTimeout(() => {
    const success = Math.random() > 0.5;
    if (success) {
      resolve("成功！");   // fulfilled に遷移
    } else {
      reject("失敗...");  // rejected に遷移
    }
  }, 1000);
});

// Promise の消費
promise
  .then((result) => {
    console.log("成功：", result); // → "成功！"
    return result + " 次の処理"; // 次の then に渡せる
  })
  .then((result) => console.log(result)) // → "成功！ 次の処理"
  .catch((error) => {
    console.log("エラー：", error); // → "失敗..."
  })
  .finally(() => {
    console.log("成功・失敗に関わらず実行");
  });

async/await（ES2017〜）

// async 関数：常に Promise を返す
async function fetchData() {
  // await：Promise の解決を待つ（同期的に書ける）
  const result = await someAsyncOperation();
  console.log(result);
  return result;
}

// try/catch でエラーハンドリング
async function loadUser(id) {
  try {
    const user = await getUser(id);
    const posts = await getPosts(user.id);
    return { user, posts };
  } catch (error) {
    console.log("エラー：", error);
  } finally {
    console.log("処理完了");
  }
}

// Promise チェーンとの比較
// Promise チェーン
getUser(1)
  .then(user => getPosts(user.id))
  .then(posts => console.log(posts))
  .catch(err => console.log(err));

// async/await（同じ処理）
async function main() {
  const user  = await getUser(1);
  const posts = await getPosts(user.id);
  console.log(posts);
}

fetch API

// fetch：HTTP リクエストを送る
async function fetchUser(id) {
  try {
    // GET リクエスト
    const response = await fetch(`https://api.example.com/users/${id}`);

    // HTTP ステータスの確認
    if (!response.ok) {
      throw new Error(`HTTP error: ${response.status}`);
    }

    // JSON として解析
    const data = await response.json();
    console.log(data);
    return data;
  } catch (error) {
    console.log("エラー：", error);
  }
}

// POST リクエスト
async function createUser(userData) {
  const response = await fetch("https://api.example.com/users", {
    method:  "POST",
    headers: { "Content-Type": "application/json" },
    body:    JSON.stringify(userData)
  });
  return response.json();
}

Promise.all / Promise.race

// Promise.all：すべての Promise が解決するまで待つ
async function loadAll() {
  const [users, posts, comments] = await Promise.all([
    fetch("/api/users").then(r => r.json()),
    fetch("/api/posts").then(r => r.json()),
    fetch("/api/comments").then(r => r.json()),
  ]);
  // 3つのリクエストを並行実行（直列より速い）
  console.log(users, posts, comments);
}

// Promise.allSettled：失敗しても全結果を取得
const results = await Promise.allSettled([p1, p2, p3]);
results.forEach(r => {
  if (r.status === "fulfilled") console.log(r.value);
  if (r.status === "rejected")  console.log(r.reason);
});

// Promise.race：最初に解決した Promise の結果を使う
const fastest = await Promise.race([
  fetch("/api/server1"),
  fetch("/api/server2"),
]);
// 最初に応答したサーバーの結果を使う

📌 まとめ

機能構文ポイント
イベント登録 el.addEventListener("click", fn) 複数登録可・removeEventListener で解除
イベントオブジェクト e.target / e.type / e.key イベントの詳細情報を持つ
伝播停止 e.stopPropagation() バブリングを止める
デフォルト動作キャンセル e.preventDefault() リンク遷移・フォーム送信などをキャンセル
遅延実行 setTimeout(fn, ms) 指定時間後に1回実行
繰り返し実行 setInterval(fn, ms) 指定間隔で繰り返し実行
Promise new Promise((resolve, reject) => {}) 非同期処理の結果を表すオブジェクト
async/await async function / await promise Promise を同期的に書ける
fetch API await fetch(url) HTTP リクエストを送る
並行処理 Promise.all([p1, p2, ...]) 複数の Promise を並行実行
