🔀 条件分岐：if / switch

if / else if / else

const score = 75;

if (score >= 90) {
  console.log('A');
} else if (score >= 70) {
  console.log('B'); // ← 実行される
} else if (score >= 50) {
  console.log('C');
} else {
  console.log('D');
}

switch 文とフォールスルー

switch 文は break を書かないと次の case に処理が フォールスルー（fall through） します。試験頻出の注意点です。 

const day = 2;

switch (day) {
  case 1:
    console.log('月曜');
    break;           // ← breakで次のcaseへ落ちない
  case 2:
    console.log('火曜'); // ← 実行
    // break がない！→ フォールスルー
  case 3:
    console.log('水曜'); // ← フォールスルーで実行される
    break;
  default:
    console.log('その他');
}
// 出力: "火曜" → "水曜"

// フォールスルーを意図的に使う例（複数caseで同じ処理）
switch (day) {
  case 6:
  case 7:
    console.log('週末'); // 6と7で同じ処理
    break;
  default:
    console.log('平日');
}
⚠️ 試験頻出： switch 文の case は 厳密等価（===） で比較します。 break を忘れると意図しないフォールスルーが発生します。 default は最後でなくても記述できますが、通常は最後に書きます。

🔁 ループ：while / for

while / do...while

// while：条件が true の間繰り返す（0回実行の可能性あり）
let i = 0;
while (i < 3) {
  console.log(i); // 0, 1, 2
  i++;
}

// do...while：必ず1回は実行してから条件チェック
let j = 10;
do {
  console.log(j); // 10（条件falseでも1回実行）
  j++;
} while (j < 3);

for / for...in / for...of

// for：カウンタ制御ループ
for (let i = 0; i < 3; i++) {
  console.log(i); // 0, 1, 2
}

// for...in：オブジェクトのキー（プロパティ名）を列挙
const obj = { a: 1, b: 2, c: 3 };
for (const key in obj) {
  console.log(key, obj[key]); // "a" 1, "b" 2, "c" 3
}

// for...of：イテラブル（配列・文字列・Map・Set等）の値を列挙
const arr = [10, 20, 30];
for (const val of arr) {
  console.log(val); // 10, 20, 30
}

// 文字列にも for...of が使える
for (const ch of 'hello') {
  console.log(ch); // h, e, l, l, o
}
構文対象取得できるもの注意点
for...in オブジェクト キー（プロパティ名） プロトタイプのプロパティも列挙される場合あり
for...of イテラブル（配列・文字列・Map・Set） 値 通常のオブジェクトには使えない

break / continue（ラベル付き）

// break：ループを抜ける
for (let i = 0; i < 5; i++) {
  if (i === 3) break;
  console.log(i); // 0, 1, 2
}

// continue：今回の繰り返しをスキップ
for (let i = 0; i < 5; i++) {
  if (i === 2) continue;
  console.log(i); // 0, 1, 3, 4
}

// ラベル付き break（ネストしたループを一気に抜ける）
outer: for (let i = 0; i < 3; i++) {
  for (let j = 0; j < 3; j++) {
    if (i === 1 && j === 1) break outer;
    console.log(i, j); // (0,0)(0,1)(0,2)(1,0) で終了
  }
}

⚠️ 例外処理：try / catch / finally / throw

// 基本構文
try {
  // 例外が発生する可能性のあるコード
  const result = JSON.parse('invalid json'); // SyntaxError発生
} catch (e) {
  // 例外が発生した場合に実行
  console.log(e.name);    // "SyntaxError"
  console.log(e.message); // エラーメッセージ
} finally {
  // 例外の有無に関わらず必ず実行
  console.log('クリーンアップ処理');
}

// throw：例外を意図的に投げる
function divide(a, b) {
  if (b === 0) {
    throw new RangeError('0で割ることはできません');
  }
  return a / b;
}

try {
  console.log(divide(10, 0));
} catch (e) {
  console.log(e instanceof RangeError); // true
  console.log(e.message); // "0で割ることはできません"
}

💡 試験ポイント：finally の実行タイミング
finally ブロックは try / catch の return 文より後に実行されます。 try 内で return しても finally は必ず実行されます。 また catch 節は省略可能（ES2019〜 try {} finally {} の形も可）。 
// finally は return より後に実行される
function test() {
  try {
    return 'try';
  } finally {
    console.log('finally実行'); // ← returnの前に実行
  }
}
console.log(test()); // "finally実行" → "try"

// エラーの種類を判別する
try {
  null.property;
} catch (e) {
  if (e instanceof TypeError) {
    console.log('型エラー');
  } else if (e instanceof ReferenceError) {
    console.log('参照エラー');
  }
}
