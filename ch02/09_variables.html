① CSS カスタムプロパティ（変数）

重要ポイント：CSS カスタムプロパティは --変数名: 値 で定義し、 var(--変数名) で参照します。 :root に定義するとグローバルに使用でき、 JavaScript からも読み書きできます。 
/* カスタムプロパティの定義 */
:root {
  /* グローバル変数（全要素から参照可能）*/
  --color-primary:   #553c9a;
  --color-secondary: #2c7a7b;
  --color-accent:    #f6ad55;
  --font-size-base:  16px;
  --spacing-md:      16px;
  --radius-md:       8px;
  --shadow-md:       0 4px 12px rgba(0,0,0,0.1);
}

/* var() で参照 */
.btn {
  background:    var(--color-primary);
  padding:       var(--spacing-md);
  border-radius: var(--radius-md);
  box-shadow:    var(--shadow-md);
}

/* フォールバック値（変数が未定義の場合）*/
.box {
  color: var(--color-text, #2d3748);
  /* --color-text が未定義なら #2d3748 を使用 */
}

/* フォールバックのネスト */
.box {
  color: var(--color-text, var(--color-primary, #553c9a));
}

② スコープと継承

重要ポイント：CSS 変数はスコープを持ちます。 :root で定義するとグローバル、 特定のセレクタで定義するとそのサブツリーのみに適用されます。 子要素は親の変数を継承します。 
/* グローバルスコープ */
:root {
  --color: purple;
}

/* ローカルスコープ（コンポーネント内で上書き）*/
.card {
  --color: teal; /* .card 内では teal が使われる */
}
.card p {
  color: var(--color); /* → teal（親から継承）*/
}
p {
  color: var(--color); /* → purple（:root から）*/
}

/* コンポーネントの変数設計パターン */
.button {
  /* コンポーネントレベルの変数 */
  --btn-bg:     var(--color-primary, #553c9a);
  --btn-color:  #fff;
  --btn-radius: var(--radius-md, 8px);

  background:    var(--btn-bg);
  color:         var(--btn-color);
  border-radius: var(--btn-radius);
}
/* 使う側が --btn-bg だけ変えれば色が変わる */
.button.danger { --btn-bg: #e53e3e; }
.button.success{ --btn-bg: #38a169; }

/* :is() / :where() との組み合わせ */
:is(h1, h2, h3) {
  color: var(--color-heading, #1a365d);
}

③ calc() 関数

重要ポイント：calc() は異なる単位を混在させた計算ができます。 CSS 変数と組み合わせることで柔軟なレイアウトを実現できます。 演算子の前後にスペースが必要です（特に + と -）。 
/* calc()：異なる単位の計算 */
.sidebar {
  width: calc(100% - 300px);
  /* 親幅100% から 300px を引いた幅 */
}

.full-bleed {
  margin-left:  calc(-1 * var(--spacing-md));
  margin-right: calc(-1 * var(--spacing-md));
  /* コンテナからはみ出す */
}

/* CSS 変数との組み合わせ */
:root { --base: 16px; }
h1 { font-size: calc(var(--base) * 2); }   /* 32px */
h2 { font-size: calc(var(--base) * 1.5); } /* 24px */

/* ⚠️ + と - の前後にスペースが必要 */
.ok  { width: calc(100% - 20px); }  /* ✅ */
.ng  { width: calc(100%-20px); }   /* ❌ スペースなし */

/* ネストした calc() */
.box {
  width: calc(calc(100% - 40px) / 3);
  /* または単純に */
  width: calc((100% - 40px) / 3);
}

④ min() / max() / clamp()

min() / max()

/* min()：引数の最小値を返す */
.container {
  width: min(100%, 1200px);
  /* 100% と 1200px の小さいほう */
  /* → 画面が1200px以下なら100%、以上なら1200px */
}

.icon {
  font-size: min(5vw, 48px);
  /* 5vw が 48px を超えないようにする */
}

/* max()：引数の最大値を返す */
.text {
  font-size: max(1rem, 2vw);
  /* 1rem と 2vw の大きいほう */
  /* → 最小でも 1rem を保証 */
}

.padding {
  padding: max(16px, 3vw);
  /* 最低16pxのパディングを保証 */
}

/* 複数の引数も可能 */
.box { width: min(300px, 50%, 80vw); }

clamp()

/* clamp(最小値, 推奨値, 最大値) */
h1 {
  font-size: clamp(1.5rem, 4vw, 3rem);
  /* 最小1.5rem・推奨4vw・最大3rem */
}

.container {
  width: clamp(320px, 90%, 1200px);
  /* 最小320px・推奨90%・最大1200px */
}

/* CSS 変数との組み合わせ */
:root {
  --fluid-h1: clamp(1.8rem, 4vw + 1rem, 3.5rem);
  --fluid-p:  clamp(1rem,  1.5vw + 0.5rem, 1.25rem);
}
h1 { font-size: var(--fluid-h1); }
p  { font-size: var(--fluid-p); }

/* clamp は min(max(最小値, 推奨値), 最大値) と同等 */
/* clamp(MIN, VAL, MAX) = min(MAX, max(MIN, VAL)) */

CSS 変数 × 数学関数の実践パターン

/* デザインシステムの構築例 */
:root {
  /* スペーシングスケール */
  --space-1: 0.25rem;   /* 4px */
  --space-2: 0.5rem;    /* 8px */
  --space-4: 1rem;      /* 16px */
  --space-8: 2rem;      /* 32px */

  /* 流体タイポグラフィ */
  --text-sm: clamp(0.75rem, 1.5vw, 0.875rem);
  --text-md: clamp(1rem,    2vw,   1.125rem);
  --text-lg: clamp(1.25rem, 3vw,   1.5rem);
  --text-xl: clamp(1.5rem,  4vw,   2rem);
  --text-2xl:clamp(2rem,    5vw,   3rem);

  /* レスポンシブコンテナ */
  --container: min(90%, 1200px);
  --container-sm: min(90%, 640px);
}

.container {
  width:  var(--container);
  margin: 0 auto;
}

/* 動的なグリッドカラム数 */
.auto-grid {
  display: grid;
  grid-template-columns:
    repeat(auto-fill, minmax(min(100%, var(--min-col-width, 200px)), 1fr));
}

⑤ ダークモード対応

重要ポイント：CSS 変数と prefers-color-scheme メディアクエリを 組み合わせることで、ダークモードを効率よく実装できます。 変数の値を切り替えるだけで全体のテーマが変わります。 
/* ダークモード対応の CSS 変数パターン */
:root {
  /* ライトモード（デフォルト）*/
  --bg:       #ffffff;
  --bg-card:  #f7fafc;
  --text:     #2d3748;
  --text-sub: #718096;
  --border:   #e2e8f0;
  --primary:  #553c9a;
}

@media (prefers-color-scheme: dark) {
  :root {
    /* ダークモード：変数の値だけ変える */
    --bg:       #1a202c;
    --bg-card:  #2d3748;
    --text:     #e2e8f0;
    --text-sub: #a0aec0;
    --border:   #4a5568;
    --primary:  #9f7aea;
  }
}

/* data-theme 属性でも切り替え可能 */
[data-theme="dark"] {
  --bg:    #1a202c;
  --text:  #e2e8f0;
}

/* JavaScript でテーマを切り替え */
document.documentElement.setAttribute('data-theme', 'dark');

⑥ JavaScript との連携

/* JavaScript から CSS 変数を読み書き */

/* 読み取り */
const root  = document.documentElement;
const style = getComputedStyle(root);
const color = style.getPropertyValue('--color-primary').trim();
// → '#553c9a'

/* 書き込み */
root.style.setProperty('--color-primary', '#e53e3e');

/* 削除（デフォルト値に戻す）*/
root.style.removeProperty('--color-primary');

/* 特定要素の変数を変更 */
const card = document.querySelector('.card');
card.style.setProperty('--card-bg', '#fef3c7');

/* アニメーションと組み合わせ */
let hue = 0;
function animateHue() {
  hue = (hue + 1) % 360;
  root.style.setProperty('--hue', hue);
  requestAnimationFrame(animateHue);
}
