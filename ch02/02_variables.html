📦 変数・定数の宣言：var / let / const

JavaScriptには変数・定数を宣言するキーワードが3つあります。現代のコードでは let / const を優先して使います。

キーワード再宣言再代入スコープ巻き上げ（Hoisting）推奨
var ✅ 可✅ 可 関数スコープ 宣言のみ巻き上げ（undefined） ❌ 非推奨
let ❌ 不可✅ 可 ブロックスコープ TDZ（一時的デッドゾーン） ✅ 推奨
const ❌ 不可❌ 不可 ブロックスコープ TDZ（一時的デッドゾーン） ✅ 最推奨 
// var：関数スコープ・再宣言可・巻き上げあり
var x = 10;
var x = 20;  // 再宣言OK（バグの温床）
console.log(x); // 20

// let：ブロックスコープ・再宣言不可
let y = 10;
// let y = 20; // SyntaxError: 再宣言不可
y = 20;  // 再代入はOK

// const：再宣言・再代入ともに不可
const PI = 3.14159;
// PI = 3; // TypeError: 再代入不可

// ただしconstのオブジェクト・配列の中身は変更可
const obj = { name: 'Alice' };
obj.name = 'Bob';  // OK（プロパティの変更は可）
// obj = {}; // TypeError: 変数自体への再代入は不可

巻き上げ（Hoisting）と TDZ

// var の巻き上げ：宣言が先頭に移動し undefined で初期化
console.log(a); // undefined（エラーにならない）
var a = 5;

// let/const の TDZ（Temporal Dead Zone）
// console.log(b); // ReferenceError: Cannot access 'b' before initialization
let b = 5;
💡 試験ポイント： let と const も巻き上げは発生しますが、TDZ（一時的デッドゾーン）により 宣言前にアクセスすると ReferenceError になります。 var のように undefined にはなりません。

🔢 データ型（Types）

JavaScriptの型は プリミティブ型（7種）と オブジェクト型 に大別されます。

分類型名例typeof の結果
プリミティブnumber42, 3.14, NaN, Infinity"number"
string"hello", 'world', `template`"string"
booleantrue, false"boolean"
nullnull"object" ⚠️
undefinedundefined"undefined"
symbolSymbol('id')"symbol"
bigint9007199254740991n"bigint"
オブジェクトobject{}, [], function(){}"object" / "function"
⚠️ 有名なバグ： typeof null は "object" を返します。これはJavaScript初期の仕様バグで、 後方互換性のため修正されていません。null チェックには value === null を使いましょう。

特殊な数値：NaN / Infinity

// NaN (Not a Number)：数値演算が失敗した結果
console.log(0 / 0);          // NaN
console.log(parseInt('abc')); // NaN
console.log(typeof NaN);      // "number"（NaNはnumber型！）

// NaN は自分自身と等しくない（唯一の例外）
console.log(NaN === NaN);    // false
console.log(isNaN(NaN));      // true
console.log(Number.isNaN(NaN)); // true（より厳密）

// Infinity
console.log(1 / 0);           // Infinity
console.log(-1 / 0);          // -Infinity
console.log(isFinite(Infinity)); // false

🔄 型変換（Type Conversion）

暗黙の型変換（型強制）

// 文字列との + は文字列結合
console.log('3' + 4);    // "34"（文字列）
console.log(3 + '4');    // "34"（文字列）

// - * / は数値に変換して計算
console.log('6' - 2);    // 4（数値）
console.log('3' * '2');   // 6（数値）

// == は型変換あり、=== は型変換なし（厳密比較）
console.log(0 ==  '');   // true（型変換あり）
console.log(0 === '');   // false（型変換なし）
console.log(null == undefined);  // true
console.log(null === undefined); // false

明示的な型変換

// → 数値へ変換
console.log(Number('42'));    // 42
console.log(Number(''));      // 0
console.log(Number(true));    // 1
console.log(Number(false));   // 0
console.log(Number(null));    // 0
console.log(Number(undefined)); // NaN
console.log(parseInt('42px')); // 42（先頭の数値部分のみ）

// → 文字列へ変換
console.log(String(42));      // "42"
console.log((42).toString()); // "42"
console.log((255).toString(16)); // "ff"（16進数）

// → 真偽値へ変換（Falsy値に注意）
console.log(Boolean(0));      // false（Falsy）
console.log(Boolean(''));     // false（Falsy）
console.log(Boolean(null));   // false（Falsy）
console.log(Boolean('0'));    // true（文字列は空でなければTruthy）
console.log(Boolean([]));     // true（空配列もTruthy）

Falsy 値（false と評価される値）

Falsy 値説明
falseboolean の false
0 / -0 / 0n数値のゼロ・BigIntのゼロ
"" / '' / ``空文字列
null値が存在しない
undefined未定義
NaN非数値

🔭 変数のスコープ

スコープとは変数が参照できる有効範囲のことです。

スコープ種別説明対象キーワード
グローバルスコープどこからでも参照可能var / let / const（関数・ブロック外）
関数スコープ関数内でのみ参照可能var / let / const（関数内）
ブロックスコープ{ } ブロック内でのみ参照可能let / const のみ
 
var   globalVar = 'global';
let   globalLet = 'global';

function outer() {
  var funcVar = 'function';
  let funcLet = 'function';

  if (true) {
    var   blockVar = 'block'; // varは関数スコープ→outer全体で有効
    let   blockLet = 'block'; // letはブロックスコープ→{}内のみ
    const blockConst = 'block';
  }

  console.log(blockVar);  // "block"（varはブロック外でも有効）
  // console.log(blockLet); // ReferenceError（ブロック外は参照不可）
}

// console.log(funcVar); // ReferenceError（関数外は参照不可）
