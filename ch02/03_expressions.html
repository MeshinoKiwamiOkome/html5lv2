// インクリメント前置 vs 後置の違い
let a = 5;

⚙️ ビット演算子

整数を 32ビットの2進数 として扱い、ビット単位で演算します。

演算子名称説明例（5 op 3）結果
&AND両方1のビットが15 & 31
|ORどちらか1のビットが15 | 37
^XORどちらか一方だけ1のビットが15 ^ 36
~NOTビットを反転（単項）~5-6
<<左シフトビットを左にずらす（×2）5 << 110
>>右シフト（符号あり）ビットを右にずらす（÷2）5 >> 12
>>>右シフト（符号なし）符号ビットも0で埋める-1 >>> 04294967295
 
// 5 = 0101, 3 = 0011 (2進数)
console.log(5 &  3); // 1  (0001) AND
console.log(5 |  3); // 7  (0111) OR
console.log(5 ^  3); // 6  (0110) XOR
console.log(~5);     // -6       NOT（~n = -(n+1)）
console.log(5 << 1); // 10 (1010) 左シフト
console.log(5 >> 1); // 2  (0010) 右シフト

// ビット演算の実用例：フラグ管理
const READ    = 0b001; // 1
const WRITE   = 0b010; // 2
const EXECUTE = 0b100; // 4
let perm = READ | WRITE;  // 3（読み書き権限）
console.log((perm & READ) !== 0);    // true（読み取り権限あり）
console.log((perm & EXECUTE) !== 0); // false（実行権限なし）

🔗 メンバ演算子・条件演算子・その他

メンバ演算子

const user = { name: 'Alice', address: { city: 'Tokyo' } };

// ドット記法
console.log(user.name);            // "Alice"

// ブラケット記法（動的なキー名に対応）
const key = 'name';
console.log(user[key]);            // "Alice"

// オプショナルチェーン ?. （ES2020）
// プロパティが存在しない場合 undefined を返す（エラーにならない）
console.log(user?.address?.city);  // "Tokyo"
console.log(user?.phone?.number);  // undefined（エラーなし）

// メソッドへのオプショナルチェーン
console.log(user?.greet?.());       // undefined（メソッドが存在しなくてもOK）

条件演算子（三項演算子）

// 構文: 条件 ? 真の値 : 偽の値
const age = 20;
const label = age >= 18 ? '成人' : '未成年';
console.log(label); // "成人"

// ネストも可能（可読性に注意）
const score = 75;
const grade = score >= 90 ? 'A'
            : score >= 70 ? 'B'
            : score >= 50 ? 'C' : 'D';
console.log(grade); // "B"

typeof / instanceof / in / delete

// typeof：型を文字列で返す
console.log(typeof 42);        // "number"
console.log(typeof 'hello');   // "string"
console.log(typeof undefined); // "undefined"
console.log(typeof null);      // "object" ← 有名なバグ

// instanceof：プロトタイプチェーンを確認
console.log([] instanceof Array);   // true
console.log([] instanceof Object);  // true（ArrayはObjectを継承）

// in：プロパティの存在確認
const obj = { x: 1, y: 2 };
console.log('x' in obj);  // true
console.log('z' in obj);  // false

// delete：プロパティの削除
delete obj.x;
console.log(obj); // { y: 2 }

🔀 論理演算子と短絡評価

演算子名称説明短絡評価
&&AND 左が Truthy なら右を返す、Falsy なら左を返す 左が Falsy → 右を評価しない
||OR 左が Truthy なら左を返す、Falsy なら右を返す 左が Truthy → 右を評価しない
??Nullish Coalescing 左が null / undefined のときだけ右を返す 左が null/undefined 以外 → 右を評価しない
!NOT 真偽値を反転 - 
// && の短絡評価
console.log(null  && 'hello'); // null（左がFalsyなので左を返す）
console.log('hi'  && 'hello'); // "hello"（左がTruthyなので右を返す）

// || の短絡評価（デフォルト値に使われる）
console.log(null  || 'default'); // "default"
console.log(0     || 'default'); // "default"（0はFalsy！）
console.log('hi'  || 'default'); // "hi"

// ?? Nullish Coalescing（ES2020）
// null / undefined のときだけ右辺を使う（0や""は通さない）
console.log(null      ?? 'default'); // "default"
console.log(undefined ?? 'default'); // "default"
console.log(0         ?? 'default'); // 0（0はnull/undefinedではない！）
console.log(''        ?? 'default'); // ""（空文字もnull/undefinedではない）
💡 試験ポイント：|| と ?? の違い
|| は Falsy（0, "", false, null, undefined, NaN） のときに右辺を返します。
?? は null / undefined のみ のときに右辺を返します。
数値の 0 や空文字 "" を有効な値として扱いたい場合は ?? を使いましょう。
