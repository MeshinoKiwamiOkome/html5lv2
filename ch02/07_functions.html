🔧 関数定義の4方式

方式構文巻き上げthisarguments
function 文 function fn() {} ✅ 完全に巻き上げ 呼び出し時に決定 ✅ あり
関数式 const fn = function() {} ❌ 変数のみ巻き上げ 呼び出し時に決定 ✅ あり
アロー関数 const fn = () => {} ❌ 変数のみ巻き上げ 定義時の外側の this ❌ なし
Function コンストラクタ new Function('x','return x') ❌ グローバル ✅ あり 
// ① function 文（宣言）：巻き上げあり
hello(); // "Hello!"（宣言前でも呼べる）
function hello() { return 'Hello!'; }

// ② 関数式：変数に代入
const greet = function(name) {
  return `Hi, ${name}`;
};

// ③ アロー関数：thisを持たない
const add = (a, b) => a + b;       // 1行なら{}とreturn省略可
const double = n => n * 2;          // 引数1つなら()省略可
const getObj = () => ({ key: 'val' }); // オブジェクトは()で包む

// ④ Function コンストラクタ（非推奨）
const mul = new Function('a', 'b', 'return a * b');
console.log(mul(3, 4)); // 12

デフォルト引数・残余引数

// デフォルト引数（ES6）
function greet(name = 'Guest', msg = 'Hello') {
  return `${msg}, ${name}!`;
}
console.log(greet());              // "Hello, Guest!"
console.log(greet('Alice'));       // "Hello, Alice!"

// 残余引数（Rest Parameters）
function sum(...nums) {
  return nums.reduce((acc, n) => acc + n, 0);
}
console.log(sum(1, 2, 3, 4)); // 10

// arguments オブジェクト（通常関数のみ）
function showArgs() {
  console.log([...arguments]); // 全引数を配列に変換
}
showArgs(1, 2, 3); // [1, 2, 3]

即時実行関数（IIFE）

// IIFE（Immediately Invoked Function Expression）
// 定義と同時に実行。スコープを汚染しない
(function() {
  const secret = '外から見えない';
  console.log(secret);
})();

// アロー関数版 IIFE
((x) => {
  console.log(x * 2);
})(21); // 42

🎯 this の挙動

this の値は関数の呼び出し方によって決まります。 アロー関数だけは例外で、定義時の外側のスコープの this を引き継ぎます。

呼び出し方通常関数の thisアロー関数の this
メソッド呼び出し obj.fn()obj定義時の外側
関数呼び出し fn()undefined（strict）/ global定義時の外側
コンストラクタ new fn()新しいインスタンス使用不可
call / apply / bind指定したオブジェクト変更不可
 
const obj = {
  name: 'Alice',

  // 通常関数：thisはobjを指す
  normalFn() {
    console.log(this.name); // "Alice"

    // 内側の通常関数：thisがundefined（strict）になる
    const inner = function() {
      console.log(this); // undefined（strict mode）
    };
    inner();
  },

  // アロー関数：thisは定義時の外側（obj）を引き継ぐ
  arrowFn() {
    const inner = () => {
      console.log(this.name); // "Alice"（外側のthisを引き継ぐ）
    };
    inner();
  }
};

// setTimeout でのアロー関数の活用
function Timer() {
  this.count = 0;
  setInterval(() => {   // アロー関数でthisを維持
    this.count++;
    console.log(this.count);
  }, 1000);
}

🔗 call / apply / bind

関数の this を明示的に指定するメソッドです。

メソッド構文引数の渡し方実行タイミング
call fn.call(thisArg, a, b) カンマ区切りで個別に渡す 即時実行
apply fn.apply(thisArg, [a, b]) 配列でまとめて渡す 即時実行
bind fn.bind(thisArg, a) カンマ区切り（部分適用も可） 新しい関数を返す（遅延実行） 
function introduce(greeting, punctuation) {
  return `${greeting}, I'm ${this.name}${punctuation}`;
}

const alice = { name: 'Alice' };
const bob   = { name: 'Bob'   };

// call：引数を個別に渡して即時実行
console.log(introduce.call(alice, 'Hello', '!'));
// "Hello, I'm Alice!"

// apply：引数を配列で渡して即時実行
console.log(introduce.apply(bob, ['Hi', '.']));
// "Hi, I'm Bob."

// bind：thisを固定した新しい関数を返す
const aliceIntro = introduce.bind(alice, 'Hey');
console.log(aliceIntro('?'));
// "Hey, I'm Alice?"

// apply の活用：配列の最大値
const nums = [3, 1, 4, 1, 5, 9];
console.log(Math.max.apply(null, nums)); // 9
console.log(Math.max(...nums));           // 9（スプレッド構文で同じ結果）

⚡ ジェネレータ関数（function* / yield）

ジェネレータ関数は yield で処理を一時停止し、 next() が呼ばれるたびに再開する特殊な関数です。 

// ジェネレータ関数の定義（function* または function *）
function* counter(start = 1) {
  console.log('開始');
  yield start;       // ① ここで一時停止
  console.log('再開1');
  yield start + 1;   // ② ここで一時停止
  console.log('再開2');
  yield start + 2;   // ③ ここで一時停止
  console.log('終了');
}

const gen = counter(10); // ジェネレータオブジェクトを生成（まだ実行されない）

console.log(gen.next()); // "開始" → { value: 10, done: false }
console.log(gen.next()); // "再開1" → { value: 11, done: false }
console.log(gen.next()); // "再開2" → { value: 12, done: false }
console.log(gen.next()); // "終了" → { value: undefined, done: true }

// for...of でジェネレータを反復
for (const val of counter(1)) {
  console.log(val); // 1, 2, 3
}

// 無限シーケンスの生成
function* infiniteId() {
  let id = 1;
  while (true) yield id++;
}
const ids = infiniteId();
console.log(ids.next().value); // 1
console.log(ids.next().value); // 2
💡 試験ポイント：ジェネレータの next() の戻り値
next() は { value: 値, done: boolean } オブジェクトを返します。 done: false は yield で一時停止中、done: true は関数が終了したことを示します。 ジェネレータ関数を呼び出しただけでは実行されず、最初の next() で開始します。

高階関数・コールバック

// 高階関数：関数を引数に取る / 関数を返す
function repeat(fn, times) {
  for (let i = 0; i < times; i++) fn(i);
}
repeat(i => console.log(`step ${i}`), 3);

// 関数を返す（カリー化）
function multiply(factor) {
  return (n) => n * factor;
}
const triple = multiply(3);
console.log(triple(7)); // 21

✏️ 演習問題

Q1 アロー関数の this について空欄を埋めてください。（穴埋め）

アロー関数は自身の this を持たず、 時の外側のスコープの this を引き継ぐ。 そのため call / apply / bind で this を することはできない。 また arguments オブジェクトも 。
