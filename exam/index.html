let currentMode = null;
let currentChapter = null;
let examQuestions = [];

function selectMode(mode) {
  currentMode = mode;
  document.querySelectorAll('.mode-card').forEach(c => c.classList.remove('selected'));
  document.querySelector(`.mode-card[data-mode="${mode}"]`).classList.add('selected');
  document.getElementById('chapter-select-wrap').style.display = mode === 'chapter' ? 'block' : 'none';
  document.getElementById('btn-start').disabled = false;
}

function startExam() {
  if (!currentMode) return;

  let pool = [...ALL_QUESTIONS];

  if (currentMode === 'chapter') {
    currentChapter = parseInt(document.getElementById('chapter-select').value);
    pool = ALL_QUESTIONS.filter(q => q.chapter === currentChapter);
    examQuestions = shuffle(pool).slice(0, Math.min(20, pool.length));
    timeLeft = 30 * 60;
  } else if (currentMode === 'real') {
    examQuestions = shuffle(pool).slice(0, 60);
    timeLeft = 90 * 60;
  } else if (currentMode === 'practice') {
    examQuestions = shuffle(pool).slice(0, 60);
    timeLeft = null;
  } else if (currentMode === 'weak') {
    const history = getExamHistory();
    const weakIds = Object.keys(history).filter(id => {
      const h = history[id];
      return h.total > 0 && (h.correct / h.total) < 0.6;
    });
    if (weakIds.length < 5) {
      examQuestions = shuffle(pool).slice(0, 30);
    } else {
      const weakPool = ALL_QUESTIONS.filter((_, i) => weakIds.includes(String(i)));
      const fill = shuffle(pool.filter((_, i) => !weakIds.includes(String(i))));
      examQuestions = shuffle([...weakPool, ...fill]).slice(0, 30);
    }
    timeLeft = 45 * 60;
  }

  answers = new Array(examQuestions.length).fill(null);
  flags   = new Array(examQuestions.length).fill(false);
  currentIndex = 0;
  startTime = Date.now();

  document.getElementById('screen-start').style.display = 'none';
  document.getElementById('screen-exam').style.display  = 'block';
  document.getElementById('screen-result').style.display = 'none';

  buildMapGrid();
  renderQuestion();
  if (timeLeft !== null) startTimer();
  else document.getElementById('timer').textContent = '∞';
}

function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}

/* ============================================================
   タイマー
   ============================================================ */
function startTimer() {
  updateTimerDisplay();
  timerInterval = setInterval(() => {
    timeLeft--;
    updateTimerDisplay();
    if (timeLeft <= 0) {
      clearInterval(timerInterval);
      finishExam();
    }
  }, 1000);
}

function updateTimerDisplay() {
  const el = document.getElementById('timer');
  const m = Math.floor(timeLeft / 60);
  const s = timeLeft % 60;
  el.textContent = `${String(m).padStart(2,'0')}:${String(s).padStart(2,'0')}`;
  el.className = 'timer';
  if (timeLeft <= 300) el.classList.add('danger');
  else if (timeLeft <= 600) el.classList.add('warn');
}

function renderQuestion() {
  const q = examQuestions[currentIndex];
  const total = examQuestions.length;

  document.getElementById('q-counter').textContent = `問題 ${currentIndex + 1} / ${total}`;

  const badge = document.getElementById('q-badge');
  badge.textContent = q.category;
  badge.style.background = q.color;

  document.getElementById('q-text').textContent = q.q;

  const codeEl = document.getElementById('q-code');
  if (q.code) {
    codeEl.textContent = q.code;
    codeEl.style.display = 'block';
  } else {
    codeEl.style.display = 'none';
  }

  const choicesEl = document.getElementById('choices');
  choicesEl.innerHTML = '';
  const labels = ['A', 'B', 'C', 'D'];
  q.choices.forEach((choice, i) => {
    const btn = document.createElement('button');
    btn.className = 'choice-btn';
    btn.innerHTML = `${labels[i]}${choice}`;

    if (currentMode === 'practice' && answers[currentIndex] !== null) {
      btn.disabled = true;
      if (i === q.answer) btn.classList.add('correct');
      else if (i === answers[currentIndex]) btn.classList.add('wrong');
    } else if (answers[currentIndex] === i) {
      btn.classList.add('selected');
    }

    btn.addEventListener('click', () => selectAnswer(i));
    choicesEl.appendChild(btn);
  });

  const expEl = document.getElementById('explanation');
  if (currentMode === 'practice' && answers[currentIndex] !== null) {
    expEl.innerHTML = q.explanation;
    expEl.classList.add('show');
  } else {
    expEl.classList.remove('show');
  }

  document.getElementById('btn-prev').disabled = currentIndex === 0;
  document.getElementById('btn-next').disabled = currentIndex === total - 1;
  document.getElementById('btn-flag').textContent = flags[currentIndex] ? '🚩 フラグ解除' : '🚩 フラグ';

  updateProgress();
  updateMapGrid();
}

function selectAnswer(i) {
  if (currentMode === 'practice' && answers[currentIndex] !== null) return;
  answers[currentIndex] = i;

  if (currentMode === 'practice') {
    renderQuestion();
  } else {
    const btns = document.querySelectorAll('.choice-btn');
    btns.forEach((b, idx) => {
      b.classList.toggle('selected', idx === i);
    });
    updateProgress();
    updateMapGrid();
  }
}

function updateProgress() {
  const answered = answers.filter(a => a !== null).length;
  const total = examQuestions.length;
  const pct = (answered / total) * 100;
  document.getElementById('progress-fill').style.width = pct + '%';
  document.getElementById('progress-label').textContent = `${answered} / ${total} 問解答済み`;
}

function buildMapGrid() {
  const grid = document.getElementById('map-grid');
  grid.innerHTML = '';
  examQuestions.forEach((_, i) => {
    const btn = document.createElement('button');
    btn.className = 'map-btn';
    btn.textContent = i + 1;
    btn.addEventListener('click', () => {
      currentIndex = i;
      renderQuestion();
      if (mapVisible) toggleMap();
    });
    grid.appendChild(btn);
  });
}

function updateMapGrid() {
  const btns = document.querySelectorAll('.map-btn');
  btns.forEach((btn, i) => {
    btn.className = 'map-btn';
    if (i === currentIndex) btn.classList.add('current');
    else if (flags[i]) btn.classList.add('flagged');
    else if (answers[i] !== null) btn.classList.add('answered');
  });
}

function toggleMap() {
  mapVisible = !mapVisible;
  document.getElementById('map-panel').style.display = mapVisible ? 'block' : 'none';
  if (mapVisible) updateMapGrid();
}

function toggleFlag() {
  flags[currentIndex] = !flags[currentIndex];
  document.getElementById('btn-flag').textContent = flags[currentIndex] ? '🚩 フラグ解除' : '🚩 フラグ';
  updateMapGrid();
}

function confirmFinish() {
  const unanswered = answers.filter(a => a === null).length;
  const msg = unanswered > 0
    ? `未解答の問題が ${unanswered} 問あります。試験を終了して採点しますか？`
    : '試験を終了して採点しますか？';
  if (confirm(msg)) finishExam();
}

/* ============================================================
   採点・結果表示
   ============================================================ */
function finishExam() {
  if (timerInterval) clearInterval(timerInterval);

  const elapsed = Math.floor((Date.now() - startTime) / 1000);
  let correct = 0;
  const chapterStats = { 1: { correct: 0, total: 0 }, 2: { correct: 0, total: 0 }, 3: { correct: 0, total: 0 }, 4: { correct: 0, total: 0 } };

  examQuestions.forEach((q, i) => {
    const ch = q.chapter;
    chapterStats[ch].total++;
    if (answers[i] === q.answer) {
      correct++;
      chapterStats[ch].correct++;
    }
  });

  const total = examQuestions.length;
  const wrong = answers.filter((a, i) => a !== null && a !== examQuestions[i].answer).length;
  const skipped = answers.filter(a => a === null).length;
  const score = Math.round((correct / total) * 100);
  const pass = score >= 65;

  // LocalStorage 保存
  saveExamHistory(chapterStats);
  saveChapterScores(chapterStats);
  localStorage.setItem('html5lv2_ch5', '100');

  // 結果画面表示
  document.getElementById('screen-exam').style.display = 'none';
  document.getElementById('screen-result').style.display = 'block';

  const circle = document.getElementById('score-circle');
  circle.className = 'score-circle ' + (pass ? 'pass' : 'fail');
  document.getElementById('score-num').textContent = score;

  const verdict = document.getElementById('verdict');
  verdict.textContent = pass ? '✅ 合格' : '❌ 不合格';
  verdict.className = 'verdict ' + (pass ? 'pass' : 'fail');

  document.getElementById('score-detail').textContent =
    `${correct}問正解 / ${total}問中（合格ライン：65%）`;

  document.getElementById('r-correct').textContent = correct;
  document.getElementById('r-wrong').textContent   = wrong;
  document.getElementById('r-skip').textContent    = skipped;

  const em = Math.floor(elapsed / 60);
  const es = elapsed % 60;
  document.getElementById('r-time').textContent = `${em}:${String(es).padStart(2,'0')}`;

  renderCategoryBars(chapterStats);
  renderReview();
}

function renderReview() {
  const list = document.getElementById('review-list');
  list.innerHTML = '';
  const labels = ['A', 'B', 'C', 'D'];

  examQuestions.forEach((q, i) => {
    const ans = answers[i];
    const isWrong = ans !== null && ans !== q.answer;
    const isSkip  = ans === null;
    if (!isWrong && !isSkip) return;

    const item = document.createElement('div');
    item.className = 'review-item';
    item.innerHTML = `
      

        ${isWrong ? '不正解' : '未解答'}
        ${q.category}
      

      
Q${i+1}. ${q.q}

      ${q.code ? `
${escHtml(q.code)}
` : ''}
      
✅ 正解：${labels[q.answer]}. ${q.choices[q.answer]}

      ${isWrong ? `
❌ あなたの解答：${labels[ans]}. ${q.choices[ans]}
` : ''}
      
${q.explanation}
`;
    list.appendChild(item);
  });

  if (list.children.length === 0) {
    list.innerHTML = '
🎉 全問正解です！
';
  }
}

function escHtml(str) {
  return str.replace(/&/g,'&').replace(//g,'>');
}

/* ============================================================
   LocalStorage 操作
   ============================================================ */
function getExamHistory() {
  try {
    return JSON.parse(localStorage.getItem('html5lv2_exam_history') || '{}');
  } catch(e) { return {}; }
}

function saveExamHistory(chapterStats) {
  const history = getExamHistory();
  examQuestions.forEach((q, i) => {
    const key = String(ALL_QUESTIONS.indexOf(q));
    if (!history[key]) {
      history[key] = { total: 0, correct: 0, wrong: 0, chapter: q.chapter, category: q.category };
    }
    history[key].total++;
    if (answers[i] === q.answer) history[key].correct++;
    else history[key].wrong++;
  });
  localStorage.setItem('html5lv2_exam_history', JSON.stringify(history));
}

function saveChapterScores(chapterStats) {
  let stored = {};
  try {
    stored = JSON.parse(localStorage.getItem('html5lv2_chapter_scores') || '{}');
  } catch(e) {}

  [1, 2, 3, 4].forEach(ch => {
    const key = `ch${ch}`;
    if (!stored[key]) stored[key] = { correct: 0, total: 0 };
    stored[key].correct += chapterStats[ch].correct;
    stored[key].total   += chapterStats[ch].total;
  });
  localStorage.setItem('html5lv2_chapter_scores', JSON.stringify(stored));
}
