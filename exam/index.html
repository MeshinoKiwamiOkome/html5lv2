function startExam() {
  if (!currentMode) return;

  var pool = ALL_QUESTIONS.slice();

  if (currentMode === 'chapter') {
    var ch = parseInt(document.getElementById('chapter-select').value, 10);
    pool = ALL_QUESTIONS.filter(function(q) { return q.chapter === ch; });
    examQuestions = shuffle(pool).slice(0, Math.min(20, pool.length));
    timeLeft = 30 * 60;
  } else if (currentMode === 'real') {
    examQuestions = shuffle(pool).slice(0, 60);
    timeLeft = 90 * 60;
  } else if (currentMode === 'practice') {
    examQuestions = shuffle(pool).slice(0, 60);
    timeLeft = null;
  } else if (currentMode === 'weak') {
    var history = getExamHistory();
    var weakIds = Object.keys(history).filter(function(id) {
      var h = history[id];
      return h.total > 0 && (h.correct / h.total) < 0.6;
    });
    if (weakIds.length < 5) {
      examQuestions = shuffle(pool).slice(0, 30);
    } else {
      var weakPool = ALL_QUESTIONS.filter(function(q, i) { return weakIds.indexOf(String(i)) !== -1; });
      var fill     = shuffle(ALL_QUESTIONS.filter(function(q, i) { return weakIds.indexOf(String(i)) === -1; }));
      examQuestions = shuffle(weakPool.concat(fill)).slice(0, 30);
    }
    timeLeft = 45 * 60;
  }

  answers      = new Array(examQuestions.length).fill(null);
  flags        = new Array(examQuestions.length).fill(false);
  currentIndex = 0;
  startTime    = Date.now();
  mapVisible   = false;
  document.getElementById('map-panel').style.display = 'none';

  showScreen('screen-exam');
  buildMapGrid();
  renderQuestion();

  if (timeLeft !== null) {
    startTimer();
  } else {
    var t = document.getElementById('timer');
    t.textContent = '\u221e';
    t.className = 'timer';
  }
}

function shuffle(arr) {
  var a = arr.slice();
  for (var i = a.length - 1; i > 0; i--) {
    var j = Math.floor(Math.random() * (i + 1));
    var tmp = a[i]; a[i] = a[j]; a[j] = tmp;
  }
  return a;
}

function startTimer() {
  if (timerInterval) { clearInterval(timerInterval); timerInterval = null; }
  updateTimerDisplay();
  timerInterval = setInterval(function() {
    timeLeft--;
    updateTimerDisplay();
    if (timeLeft <= 0) {
      clearInterval(timerInterval);
      timerInterval = null;
      finishExam();
    }
  }, 1000);
}

function updateTimerDisplay() {
  var el = document.getElementById('timer');
  var m  = Math.floor(timeLeft / 60);
  var s  = timeLeft % 60;
  el.textContent = pad(m) + ':' + pad(s);
  el.className = 'timer';
  if      (timeLeft <= 300) el.classList.add('danger');
  else if (timeLeft <= 600) el.classList.add('warn');
}

function pad(n) { return n < 10 ? '0' + n : '' + n; }

function renderQuestion() {
  var q     = examQuestions[currentIndex];
  var total = examQuestions.length;

  document.getElementById('q-counter').textContent = '\u554f\u984c ' + (currentIndex + 1) + ' / ' + total;

  var badge = document.getElementById('q-badge');
  badge.textContent      = q.category;
  badge.style.background = q.color;

  document.getElementById('q-text').textContent = q.q;

  var codeEl = document.getElementById('q-code');
  if (q.code) {
    codeEl.textContent   = q.code;
    codeEl.style.display = 'block';
  } else {
    codeEl.style.display = 'none';
  }

  var choicesEl = document.getElementById('choices');
  choicesEl.innerHTML = '';
  var labels = ['A', 'B', 'C', 'D'];

  for (var i = 0; i < q.choices.length; i++) {
    (function(idx) {
      var btn = document.createElement('button');
      btn.className = 'choice-btn';

      var lbl = document.createElement('span');
      lbl.className   = 'choice-label';
      lbl.textContent = labels[idx];

      var txt = document.createElement('span');
      txt.textContent = q.choices[idx];

      btn.appendChild(lbl);
      btn.appendChild(txt);

      if (currentMode === 'practice' && answers[currentIndex] !== null) {
        btn.disabled = true;
        if (idx === q.answer)                   btn.classList.add('correct');
        else if (idx === answers[currentIndex]) btn.classList.add('wrong');
      } else if (answers[currentIndex] === idx) {
        btn.classList.add('selected');
      }

      btn.addEventListener('click', function() { selectAnswer(idx); });
      choicesEl.appendChild(btn);
    })(i);
  }

  var expEl = document.getElementById('explanation');
  if (currentMode === 'practice' && answers[currentIndex] !== null) {
    expEl.innerHTML = q.explanation;
    expEl.classList.add('show');
  } else {
    expEl.innerHTML = '';
    expEl.classList.remove('show');
  }

  document.getElementById('btn-prev').disabled = (currentIndex === 0);
  document.getElementById('btn-next').disabled = (currentIndex === total - 1);
  document.getElementById('btn-flag').textContent = flags[currentIndex] ? '\u2691 \u30d5\u30e9\u30b0\u89e3\u9664' : '\u2691 \u30d5\u30e9\u30b0';

  updateProgress();
  updateMapGrid();
}

function selectAnswer(i) {
  if (currentMode === 'practice' && answers[currentIndex] !== null) return;
  answers[currentIndex] = i;
  if (currentMode === 'practice') {
    renderQuestion();
  } else {
    var btns = document.querySelectorAll('.choice-btn');
    for (var k = 0; k < btns.length; k++) {
      if (k === i) btns[k].classList.add('selected');
      else         btns[k].classList.remove('selected');
    }
    updateProgress();
    updateMapGrid();
  }
}

function updateProgress() {
  var answered = 0;
  for (var i = 0; i < answers.length; i++) { if (answers[i] !== null) answered++; }
  var total = examQuestions.length;
  var pct   = (answered / total) * 100;
  document.getElementById('progress-fill').style.width  = pct + '%';
  document.getElementById('progress-label').textContent = answered + ' / ' + total + ' \u554f\u89e3\u7b54\u6e08\u307f';
}

function finishExam() {
  if (timerInterval) { clearInterval(timerInterval); timerInterval = null; }

  var elapsed = Math.floor((Date.now() - startTime) / 1000);
  var correct = 0;
  var chapterStats = {
    1: { correct: 0, total: 0 },
    2: { correct: 0, total: 0 },
    3: { correct: 0, total: 0 },
    4: { correct: 0, total: 0 }
  };

  for (var i = 0; i < examQuestions.length; i++) {
    var q  = examQuestions[i];
    var ch = q.chapter;
    chapterStats[ch].total++;
    if (answers[i] === q.answer) {
      correct++;
      chapterStats[ch].correct++;
    }
  }

  var total   = examQuestions.length;
  var wrong   = 0;
  var skipped = 0;
  for (var i = 0; i < answers.length; i++) {
    if (answers[i] === null) skipped++;
    else if (answers[i] !== examQuestions[i].answer) wrong++;
  }

  var score = Math.round((correct / total) * 100);
  var pass  = score >= 65;

  saveExamHistory(chapterStats);
  saveChapterScores(chapterStats);
  localStorage.setItem('html5lv2_ch5', '100');

  showScreen('screen-result');

  var circle = document.getElementById('score-circle');
  circle.className = 'score-circle ' + (pass ? 'pass' : 'fail');
  document.getElementById('score-num').textContent = score;

  var verdict = document.getElementById('verdict');
  verdict.textContent = pass ? '\u2705 \u5408\u683c' : '\u274c \u4e0d\u5408\u683c';
  verdict.className   = 'verdict ' + (pass ? 'pass' : 'fail');

  document.getElementById('score-detail').textContent =
    correct + '\u554f\u6b63\u89e3 / ' + total + '\u554f\u4e2d\uff08\u5408\u683c\u30e9\u30a4\u30f3\uff1a65%\uff09';

  document.getElementById('r-correct').textContent = correct;
  document.getElementById('r-wrong').textContent   = wrong;
  document.getElementById('r-skip').textContent    = skipped;
  document.getElementById('r-time').textContent    = Math.floor(elapsed / 60) + ':' + pad(elapsed % 60);

  renderCategoryBars(chapterStats);
  renderReview();
}

/* ============================================================

   ============================================================ */
function getExamHistory() {
  try   { return JSON.parse(localStorage.getItem('html5lv2_exam_history') || '{}'); }
  catch (e) { return {}; }
}

function saveExamHistory(chapterStats) {
  var history = getExamHistory();
  for (var i = 0; i < examQuestions.length; i++) {
    var q   = examQuestions[i];
    var key = String(ALL_QUESTIONS.indexOf(q));
    if (!history[key]) {
      history[key] = { total: 0, correct: 0, wrong: 0, chapter: q.chapter, category: q.category };
    }
    history[key].total++;
    if (answers[i] === q.answer) history[key].correct++;
    else                          history[key].wrong++;
  }
  localStorage.setItem('html5lv2_exam_history', JSON.stringify(history));
}

function saveChapterScores(chapterStats) {
  var stored = {};
  try { stored = JSON.parse(localStorage.getItem('html5lv2_chapter_scores') || '{}'); }
  catch (e) { stored = {}; }
  for (var ch = 1; ch <= 4; ch++) {
    var key = 'ch' + ch;
    if (!stored[key]) stored[key] = { correct: 0, total: 0 };
    stored[key].correct += chapterStats[ch].correct;
    stored[key].total   += chapterStats[ch].total;
  }
  localStorage.setItem('html5lv2_chapter_scores', JSON.stringify(stored));
}
